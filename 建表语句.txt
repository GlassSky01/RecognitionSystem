create database if not exists GlassSkyRecognitionSystem;
use GlassSkyRecognitionSystem;

create table if not exists Users(
    id int auto_increment primary key,
    username varchar(20),
    password varchar(20),
    role int
)ENGINE=InnoDB , auto_increment= 100000;

create table if not exists UsersInformation(
    id int ,
    name varchar(20),
    phoneNumber varchar(20),
    email varchar(20),
    professional varchar(20),
    class int ,
    tutor varchar(20),
    foreign key (id) references users(id)
)ENGINE=InnoDB;

create table if not exists CreditRequestMain(
    requestId int primary key auto_increment,
    tutor varchar(20) ,
    file longblob,
    dates date,
    total double,
    id int ,
    foreign key (id) references users(id) on delete cascade
)ENGINE=InnoDB;


create table if not exists CreditRequestDetails(
    types varchar(20) ,
    project varchar(20) ,
    content varchar(20) ,
    score varchar(20),
    requestId int ,
    foreign key (requestId) references CreditRequestMain(requestId) on delete cascade
)ENGINE=InnoDB;
